## Code Example: Service_Status.py (Full)

```python
print("\n=== TEST 2: DownloadManager Plugin Activation ===")
try:
    print("Attempting to activate DownloadManager plugin...")
    activation_result = ensure_plugin_active(obj, "org.rdk.DownloadManager")
    
    if activation_result:
        print("SUCCESS: DownloadManager plugin activated successfully")
        
        is_active = thunder_is_plugin_active("org.rdk.DownloadManager")
        if is_active:
            print("VERIFIED: DownloadManager plugin is confirmed active")
            tdkTestObj2 = obj.createTestStep('downloadmanager_activate')
            tdkTestObj2.setResultStatus("SUCCESS")
            print("[TEST 2 RESULT] : PASS")
        else:
            print("WARNING: Activation returned True but status check shows inactive")
            print("This may be a timing issue. Retrying...")
            time.sleep(1)
            is_active = thunder_is_plugin_active("org.rdk.DownloadManager")
            if is_active:
                print("SUCCESS: Plugin confirmed active after retry")
                tdkTestObj2 = obj.createTestStep('downloadmanager_activate')
                tdkTestObj2.setResultStatus("SUCCESS")
                print("[TEST 2 RESULT] : PASS")
            else:
                print("FAILURE: Plugin still not active after retry")
                tdkTestObj2 = obj.createTestStep('downloadmanager_activate')
                tdkTestObj2.setResultStatus("FAILURE")
                print("[TEST 2 RESULT] : FAIL")
    else:
        print("FAILURE: DownloadManager plugin activation failed")
        tdkTestObj2 = obj.createTestStep('downloadmanager_activate')
        tdkTestObj2.setResultStatus("FAILURE")
        print("[TEST 2 RESULT] : FAIL")
except Exception as e:
    print("FAILURE: Error during plugin activation: %s" % str(e))
    tdkTestObj2 = obj.createTestStep('downloadmanager_activate')
    tdkTestObj2.setResultStatus("FAILURE")
    print("[TEST 2 RESULT] : FAIL")
```

## Function Signature

```python
def ensure_plugin_active(
    tdk_obj,                          # TDK scripting library object
    callsign: str,                    # Plugin callsign (e.g., "org.rdk.DownloadManager")
    jsonrpc_url: Optional[str] = None, # JSON-RPC endpoint (optional)
    wait_seconds: float = 1.0         # Wait time before status re-check
) -> bool:                            # Returns True if active, False otherwise
    """
    Ensure a Thunder plugin is active.
    
    Tries Thunder Controller activation first,
    then falls back to JSON-RPC if provided.
    
    Returns True if plugin is active at end, else False
    """
```
